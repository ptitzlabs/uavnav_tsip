79 uint8_t validate_packet(const uint8_t *buffer, const uint32_t start,
80  const uint32_t end) {
81  uint32_t len = end - start;
83  // packet size
84  if (len < 6 || len > MAX_DATA_SIZE + 6) {
85  return size_mismatch;
86  }
87  // ID
88  if (buffer[start] == DLE) {
89  return illegal_id;
90  }
91 
92  // chksum
93  uint32_t chksum_start = end - 4;

98  }
